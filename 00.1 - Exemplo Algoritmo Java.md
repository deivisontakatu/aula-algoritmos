# ☕ Algoritmo Completo em Java (Comentado)

Este exemplo reúne os principais conceitos de algoritmos e lógica de programação utilizando Java.

---

## 💻 Código em Java

```java id="j3xk91"
// ============================================
// ☕ Exemplo Completo de Algoritmo em Java
// ============================================

// Importação necessária para entrada de dados
import java.util.Scanner;

public class AlgoritmoCompleto {

    public static void main(String[] args) {

        // --------------------------------------------
        // 📌 1. Tipos de dados e variáveis
        // --------------------------------------------

        int numeroInteiro = 10;        // tipo inteiro
        double numeroReal = 5.5;       // tipo real
        String texto = "Exemplo";      // texto (string)
        boolean valorLogico = true;    // valor lógico

        System.out.println("Valores iniciais:");
        System.out.println(numeroInteiro + " " + numeroReal + " " + texto + " " + valorLogico);

        // --------------------------------------------
        // 📌 2. Operadores
        // --------------------------------------------

        int soma = numeroInteiro + 5;
        int subtracao = numeroInteiro - 2;
        int multiplicacao = numeroInteiro * 2;
        double divisao = numeroInteiro / 2.0;

        // Operadores lógicos e relacionais
        if (numeroInteiro > 5 && valorLogico) {
            System.out.println("Condição lógica verdadeira");
        }

        // --------------------------------------------
        // 📌 3. Estrutura condicional
        // --------------------------------------------

        if (numeroInteiro < 5) {
            System.out.println("Número menor que 5");
        } else if (numeroInteiro == 10) {
            System.out.println("Número igual a 10");
        } else {
            System.out.println("Número maior que 5 e diferente de 10");
        }

        // --------------------------------------------
        // 📌 4. Estruturas de repetição
        // --------------------------------------------

        System.out.println("\nLoop FOR:");
        for (int i = 1; i <= 5; i++) {
            System.out.println("Valor de i: " + i);
        }

        System.out.println("\nLoop WHILE:");
        int contador = 1;
        while (contador <= 3) {
            System.out.println("Contador: " + contador);
            contador++; // incremento
        }

        // --------------------------------------------
        // 📌 5. Funções (métodos)
        // --------------------------------------------

        double resultadoMedia = calcularMedia(7, 8, 9);
        System.out.println("\nResultado da média: " + resultadoMedia);

        // --------------------------------------------
        // 📌 6. Vetores (arrays)
        // --------------------------------------------

        int[] vetor = {1, 2, 3, 4, 5};

        System.out.println("\nElementos do vetor:");
        for (int numero : vetor) {
            System.out.println(numero);
        }

        // --------------------------------------------
        // 📌 7. Matrizes (arrays bidimensionais)
        // --------------------------------------------

        int[][] matriz = {
            {1, 2},
            {3, 4}
        };

        System.out.println("\nElementos da matriz:");
        for (int i = 0; i < matriz.length; i++) {
            for (int j = 0; j < matriz[i].length; j++) {
                System.out.print(matriz[i][j] + " ");
            }
            System.out.println();
        }

        // --------------------------------------------
        // 📌 8. Teste simples (simulação)
        // --------------------------------------------

        testeMedia();

        // --------------------------------------------
        // 📌 9. Entrada de dados e uso completo
        // --------------------------------------------

        Scanner scanner = new Scanner(System.in);

        System.out.println("\n--- Sistema de exemplo ---");
        System.out.print("Digite um número: ");
        int valor = scanner.nextInt();

        // Verificação de número par ou ímpar
        if (valor % 2 == 0) {
            System.out.println("Número par");
        } else {
            System.out.println("Número ímpar");
        }

        // Tabuada
        System.out.println("\nTabuada:");
        for (int i = 1; i <= 10; i++) {
            System.out.println(valor + " x " + i + " = " + (valor * i));
        }

        scanner.close();
    }

    // --------------------------------------------
    // 📌 Função para calcular média
    // --------------------------------------------
    public static double calcularMedia(double a, double b, double c) {
        return (a + b + c) / 3;
    }

    // --------------------------------------------
    // 📌 Teste simples
    // --------------------------------------------
    public static void testeMedia() {
        if (calcularMedia(10, 10, 10) == 10) {
            System.out.println("\nTeste da função passou!");
        }
    }
}
```
